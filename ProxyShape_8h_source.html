125  m_selected.push_back(path);

131  inline void remove(SdfPath path)
132  {
133  auto it = std::find(m_selected.begin(), m_selected.end(), path);
134  if(it != m_selected.end())
135  {
136  m_selected.erase(it);
137  }
138  }
139 
142  inline void toggle(SdfPath path)
143  {
144  auto it = std::find(m_selected.begin(), m_selected.end(), path);
145  if(it == m_selected.end())
146  {
147  m_selected.push_back(path);
148  }
149  else
150  {
151  m_selected.erase(it);
152  }
153  }
154 
157  inline bool isSelected(const SdfPath& path) const
158  { return std::find(m_selected.begin(), m_selected.end(), path) != m_selected.end(); }

163  { return m_selected; }
